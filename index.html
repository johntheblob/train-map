<script>

const map = L.map('map').setView([57.7, 11.9], 6);


L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);


let markers = {};


// JUSTERAD KONVERTERING för dina värden

function convert(x, z) {

    // Ditt Roblox koordinatsystem (-2206 → 1508) till lat/lon

    return [

        57.7 + (z / 1000),   // lat

        11.9 + (x / 1000)    // lon

    ];

}


async function loadTrains() {

    const res = await fetch("https://train-api-7hml.onrender.com/trains");

    const data = await res.json();


    // Debug

    document.getElementById("debug").innerText = JSON.stringify(data, null, 2);


    let latlngs = []; // Lista för alla markörers position


    data.forEach(train => {

        const [lat, lon] = convert(train.x, train.z);

        latlngs.push([lat, lon]);


        if (markers[train.id]) {

            markers[train.id].setLatLng([lat, lon]);

        } else {

            markers[train.id] = L.marker([lat, lon]).addTo(map)

                .bindPopup(train.id);

        }

    });


    // Zooma/fokusera på alla markörer

    if (latlngs.length > 0) {

        const bounds = L.latLngBounds(latlngs);

        map.fitBounds(bounds, {padding: [50, 50]});

    }

}


// Kör direkt första gången

loadTrains();


// Uppdatera var 3:e sekund

setInterval(loadTrains, 3000);


</script>
